37  CVoid put(const K& key, const V& value) {
38  auto cur = cache_.find(key);
39  if (cur != cache_.end()) {
40  /* 如果在cache中找到，则删除（稍后更新至最前） */
41  nodes_.erase(cache_[key]);
42  cache_.erase(key);
43  cur_size_--;
44  } else if (cur_size_ >= capacity_) {
45  /* 如果超限，则删除最后一个节点信息 */
46  auto back = nodes_.back();
47  cache_.erase(back.key_);
48  nodes_.pop_back();
49  cur_size_--;
50  }
51 
52  nodes_.emplace_front(ULruNode<K, V>(key, value));
53  cache_[key] = nodes_.begin();
54  cur_size_++;
55  }

63  CBool get(const K& key, V& value) {
