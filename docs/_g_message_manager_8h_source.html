201  CIndex bindTopic(const std::string& topic, CUInt size) {
202  auto innerTopic = internal::PUB_SUB_PREFIX + topic;
203 
204  CGRAPH_LOCK_GUARD lk(pub_sub_mutex_);
205  auto message = CAllocator::safeMallocTemplateCObject<GMessage<TImpl>>(size);
206  CIndex connId = (++cur_conn_id_);
207  auto result = pub_sub_message_map_.find(innerTopic);
208  if (result != pub_sub_message_map_.end()) {
209  // 如果之前有的话，则在后面添加一个
210  auto& messageSet = result->second;
211  messageSet.insert((GMessagePtr<T>)message);
212  } else {
213  // 如果是这个topic第一次被绑定，则创建一个对应的set信息
214  std::set<GMessagePtr<T>> messageSet;
215  messageSet.insert((GMessagePtr<T>)message);
216  pub_sub_message_map_[innerTopic] = messageSet;
217  }
218  conn_message_map_[connId] = (GMessagePtr<T>)message;
219  return connId;
220  }
