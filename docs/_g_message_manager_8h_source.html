195  CIndex bindTopic(const std::string& topic, CUInt size) {
196  auto innerTopic = internal::PUB_SUB_PREFIX + topic;
197 
198  CGRAPH_LOCK_GUARD lk(pub_sub_mutex_);
199  auto message = CAllocator::safeMallocTemplateCObject<GMessage<TImpl>>(size);
200  CIndex connId = (++cur_conn_id_);
201  auto result = pub_sub_message_map_.find(innerTopic);
202  if (result != pub_sub_message_map_.end()) {
203  // 如果之前有的话，则在后面添加一个
204  auto& messageSet = result->second;
205  messageSet.insert((GMessagePtr<T>)message);
206  } else {
207  // 如果是这个topic第一次被绑定，则创建一个对应的set信息
208  std::set<GMessagePtr<T>> messageSet;
209  messageSet.insert((GMessagePtr<T>)message);
210  pub_sub_message_map_[innerTopic] = messageSet;
211  }
212  conn_message_map_[connId] = (GMessagePtr<T>)message;
213  return connId;
214  }
